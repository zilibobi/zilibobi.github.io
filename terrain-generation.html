    local generator = {}
    generator.__index = generator
    --
    local X,Z = 10000,100
    local OFFSET = 4
    --
    function generator:new()
        local metatable = setmetatable({},generator)
        --
        self.grid = {}
        --
        return metatable
    end
    --
    function generator:generate()
        local all = 0
        --
        game.ReplicatedStorage.Amount.Value = X
        game.ReplicatedStorage.Z.Value = Z
        --
        local rng = math.clamp(math.random() * 20,8,20) + math.random()
        --
        for x = 1,X do
            self.grid[x] = {}
            --
            for z = 1,Z do
                self.grid[x][z] = math.noise(x/rng,z/rng) * 5
            end
        end
        --
        for x = 1,X do
            for z = 1,Z do
                all += 1
                --
                local y = self.grid[x][z]
                --
                local block = script.Block:Clone()
                block.Name = all
                block.Position = Vector3.new(x * OFFSET,y,z * OFFSET)
                --
                if y > -1 then
                    block.Material = Enum.Material.Grass
                    block.Color = Color3.fromRGB(86, 207, 70)
                else
                    block.Material = Enum.Material.Sand
                    block.Color = Color3.fromRGB(253, 255, 99)
                end
                --
                block.Parent = workspace.Generated
                --
                game.ReplicatedStorage.Done.Value = z + 1
                --
                game:GetService("RunService").Heartbeat:Wait()
            end
            --
            game.ReplicatedStorage.Overall.Value = x + 1
        end
        --
        local water = script.Block:Clone()
        water.Name = "Water"
        water.CanCollide = false
        water.Transparency = 0.8
        water.Color = Color3.fromRGB(0, 0, 255)
        water.Size = Vector3.new((X - 0.001) * OFFSET,10,(Z - 0.001) * OFFSET)
        water.Position = Vector3.new(((X + 1) * OFFSET) / 2,-0.999,((Z + 1) * OFFSET) / 2)
        --
        local mesh = Instance.new("SpecialMesh")
        mesh.Scale = Vector3.new(OFFSET,1,OFFSET)
        mesh.MeshType = Enum.MeshType.Brick
        mesh.Parent = water
        --
        water.Parent = workspace.Generated
    end
