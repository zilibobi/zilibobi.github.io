Ban Module

This module is open-sourced, you can use it anywhere you want.

    local Players = game:GetService("Players")
    local DSS = game:GetService("DataStoreService")
    local DataStorage = DSS:GetDataStore("BanData")
    
    local banModule = {}
    
    local tableTemplate = {Banned = false; BanReason = "None"; BannedBy = "None";}
    tableTemplate.__index = tableTemplate
    
    function tableTemplate.new(banType, banReason, bannedBy)
        local banTable = {
            Banned = banType; 
            BanReason = banReason; 
            BannedBy = bannedBy;
        }
    
        return setmetatable(banTable, tableTemplate)
    end
    
    local function saveData(data, id)
        for x = 1,10 do
            local success, err = pcall(function()
                DataStorage:UpdateAsync(id.."_", function(oldVal)
                    local newVal = oldVal or data
                    if not newVal.Banned == false then
                        newVal.Banned = data.Banned
                        return newVal
                    end
                end)
            end)
            if success then
                print("Saved Data")
                break
            end
            if err then
                warn("Failed to load data. Attempt: "..x..". Error: "..err)
            end
            if err and x == 10 then
                return false, err
            end
        end
    end
    
    local function getData(player)
        for x = 1,10 do
            local data
            local success, err = pcall(function()
                data = DataStorage:GetAsync(player.UserId.."_")
            end)
            if data then
                if success then
                    print("Got Data For "..player.Name)
                    if data.Banned == true then
                        player:Kick("You have been banned by "..data.BannedBy..". Reason: "..data.BanReason)
                    end
                    break
                end
            end
            if not data and success then
                data = tableTemplate.new
                break
            end
            if err and x == 10 then
                return false, err
            end
        end
    end
    
    function banModule:Ban(playerName, reason, bannedBy)
        local banData = tableTemplate.new(true, reason, bannedBy)
        local id
    
        local success, err = pcall(function()
            id = Players:GetUserIdFromNameAsync(playerName)
        end)
        
        local success, err = saveData(banData, id)
        if success == nil then
            if game.Players:FindFirstChild(playerName) then
                return game.Players:FindFirstChild(playerName):Kick("You have been banned by "..bannedBy..". Reason: "..reason)
            end
        end
        if success == false then
            return "Failed to save data for "..playerName..". Error: "..err
        end
    end
    
    function banModule:UnBan(playerName)
        local banData = tableTemplate.new(false, "None", "None")
        local id
        
        local success, err = pcall(function()
            id = Players:GetUserIdFromNameAsync(playerName)
        end)
        
        if success then
            local succ, erro = saveData(banData, id)
            if succ == nil then
                return "Successfully unabnned "..playerName
            end
        end
        if success == false then
            return "Failed to save data for "..playerName.Name..". Error: "..err
        end
    end
    
    function banModule:CheckBan(plr)
        local success, err = getData(plr)
    end
    
    return banModule
